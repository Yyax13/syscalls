Exemplo: write + exit
Assemble & link (nasm + ld)
section .data
    msg db "Hello", 10
    len equ $ - msg

section .text
    global _start

_start:
    mov rax, 1
    mov rdi, 1
    mov rsi, msg
    mov rdx, len
    syscall

    mov rax, 60
    xor rdi, rdi
    syscall
Dica: use registros conforme a convenção (rdi, rsi, rdx, r10, r8, r9). Retorno em rax (-errno em caso de erro).
